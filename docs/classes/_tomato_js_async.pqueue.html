Class PQueue

异步队列

新增于v0.0.18

脚本举例

  import { PQueue, sleep } from '@tomato-js/async'
  const queue = new PQueue({
    autoStart:true, // 是否自动开始执行队列
    concurrency: 1 //队列执行并发数
  });
  let str = "";
  queue.add(async () => {
    await sleep(20);
    str = str + 1;
  });
  queue.add(async () => {
    await sleep(30);
    str = str + 2;
  });
  queue.add(async () => {
    await sleep(10);
    str = str + 3;
  });
  // 同步方法监听
  await queue.onIdle()//str=123;
  // 通过事件监听
  queue.on('idle',()=>str)//123
