    获取对象某些值，组成新对象

    脚本举例

      import { pick } from '@tomato-js/object'
  const obj = {a:{b:{c:'d'}},aa:'bb',e:'f'}
  const obj1 = pick(obj,['aa']);//{aa:'bb'}
  const obj2 = pick(obj,['aa','e']);//{aa:'bb',e:'f'}
  const obj3 = pick(obj,['aa','e'],(k,v)=>k==='e'?v+1:v);//{aa:'bb',e:'f1'}
